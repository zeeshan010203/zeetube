from pptx import Presentation from pptx.util import Inches import requests from io import BytesIO # Create a presentation object prs = Presentation() # Add a slide with title and content layout slide_layout = prs.slide_layouts[1] slide = prs.slides.add_slide(slide_layout) # Add title and content to the slide title = slide.shapes.title title.text = "Where is she from?" content = slide.placeholders[1] content.text = "Beyoncé is from Houston, Texas, USA." # Add an image of Beyoncé image_url = "https://upload.wikimedia.org/wikipedia/commons/4/44/Beyonce_The_Lion_King_European_Premiere_2019.png" response = requests.get(image_url) image_stream = BytesIO(response.content) image = slide.shapes.add_picture(image_stream, Inches(1), Inches(2), width=Inches(4.5), height=Inches(3.5)) # Save the presentation file_path = "/mnt/data/Beyonce_Where_is_she_from.pptx" prs.save(file_path) file_path
